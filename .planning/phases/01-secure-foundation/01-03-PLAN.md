it("registers an IP asset with SHA-256 hash, metadata URI, IP type, and WIPO fields", async () => {

  const perceptualHash = Buffer.alloc(32);

    program.programId
  );
  const [contentHashRegistryPda] = PublicKey.findProgramAddressSync(
    [Buffer.from("content_hash_index"), contentHash],
    program.programId
  );

  await program.methods
    .registerIp(
      Array.from(contentHash),
      Array.from(perceptualHash),
      { visualArt: {} },           // IPType

      42,                          // nice_class
      1,                           // berne_category

      ipAsset: ipAssetPda,
      contentHashRegistry: contentHashRegistryPda,
      creator: creator.publicKey,

    .rpc();

  const account = await program.account.ipAsset.fetch(ipAssetPda);
